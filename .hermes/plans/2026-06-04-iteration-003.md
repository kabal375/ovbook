```python
import re

def score_heading(c: Chunk, body_font_size: float) -> float:
    """Score a Chunk candidate for being a real heading.
    
    Positive signals: chapter keyword, numbered heading, larger font, bold.
    Negative signals: too short, TOC dots, near drawing, index letter.
    """
    s = 0.0
    text = c.heading.strip()

    # --- Negative signals ---
    if len(text) < 4:
        s -= 5
    # TOC entry: leader dots + optional page number at end
    if re.search(r"\.{3,}\s*\d*\s*$", text):
        s -= 8
    if re.match(r"\.{3,}", text):
        s -= 8  # pure dots line
    # Pure index letter (single uppercase or single char with page numbers following)
    if re.match(r"^[A-ZА-Я]$", text) or re.match(r"^[A-Z],\s*[A-Z]$", text):
        if c.level >= 2:
            s -= 8
    # Paragraph symbol or random chart artifact
    if re.match(r"^[\W_]+$", text):
        s -= 10

    # --- Positive signals ---
    if re.match(r"^(chapter|part|appendix|lecture|lesson|module)\b", text, re.IGNORECASE):
        s += 6
    if re.match(r"^\d+(\.\d+)*\s+\S+", text):
        s += 5
    if c.font_size >= body_font_size * 1.18:
        s += 3
    if c.is_bold:
        s += 2

    # --- Pre-computed flags ---
    if c.near_drawing:
        s -= 6
    if c.looks_like_toc_entry:
        s -= 10
    if c.looks_like_index_letter:
        s -= 8

    return s
```

```python
def extract_pdf(path: Path) -> list[Chunk]:
    """Extract structured chunks from a PDF using font-size heuristics and scoring."""
    from .split import Chunk, score_heading
    
    doc = fitz.open(str(path))
    body_size = _compute_body_size(doc)
    
    chunks: list[Chunk] = []
    seq = 0
    
    for page_num, page in enumerate(doc):
        blocks = page.get_text("dict")["blocks"]
        drawings = page.cluster_drawings() if hasattr(page, "cluster_drawings") else []
        
        for block in blocks:
            if block["type"] != 0:  # skip images
                continue
            
            for line in block["lines"]:
                text_parts = []
                max_size = 0.0
                is_bold = False
                
                for span in line["spans"]:
                    text_parts.append(span["text"])
                    max_size = max(max_size, span["size"])
                    if span["font"] and ("Bold" in span["font"] or "bold" in span["font"]):
                        is_bold = True
                    # Check font flags (bit 2 = bold)
                    if span.get("flags", 0) & 2:
                        is_bold = True
                
                text = "".join(text_parts).strip()
                if not text:
                    continue
                
                # Determine if near drawing
                bbox = fitz.Rect(line["bbox"]) if "bbox" in line else None
                near_drawing = _is_near_drawing(bbox, drawings) if bbox and drawings else False
                
                # Detect TOC entry
                looks_like_toc = bool(re.search(r"\.{3,}\s*\d*\s*$", text)) or bool(re.match(r"\.{3,}", text))
                
                # Detect Index letter
                looks_like_index = bool(re.match(r"^[A-ZА-Я]$", text)) or bool(re.match(r"^[A-Z],\s*[A-Z]$", text))
                
                # Scoring candidate: font >= 1.18 * body_size
                level = 3  # default body-level
                if max_size >= body_size * 1.3:
                    ratio = max_size / body_size
                    if ratio >= 2.0:
                        level = 1
                    elif ratio >= 1.6:
                        level = 2
                    else:
                        level = 3
                
                chunk = Chunk(
                    heading=text,
                    content=text,
                    level=level,
                    sequence=seq,
                    font_size=max_size,
                    is_bold=is_bold,
                    near_drawing=near_drawing,
                    looks_like_toc_entry=looks_like_toc,
                    looks_like_index_letter=looks_like_index,
                )
                
                score = score_heading(chunk, body_size)
                chunk.score = score
                chunks.append(chunk)
                seq += 1
    
    doc.close()
    return chunks
```

Also add helper functions:

```python
def _compute_body_size(doc: fitz.Document) -> float:
    """Compute body font size as median of all span sizes."""
    from collections import Counter
    sizes: list[float] = []
    for page in doc:
        blocks = page.get_text("dict")["blocks"]
        for block in blocks:
            if block["type"] == 0:
                for line in block["lines"]:
                    for span in line["spans"]:
                        sizes.append(span["size"])
        if len(sizes) > 200:
            break
    if not sizes:
        return 12.0
    # Use median instead of mode — more robust
    sizes.sort()
    return sizes[len(sizes) // 2]


def _is_near_drawing(bbox, drawings, threshold: float = 20.0) -> bool:
    """Check if bbox is near any drawing cluster."""
    if not bbox or not drawings:
        return False
    for drawing in drawings:
        d_rect = fitz.Rect(drawing) if isinstance(drawing, (list, tuple)) else drawing
        if bbox.distance_to(d_rect) < threshold:
            return True
    return False
```

```python
@dataclass
class ChapterGroup:
    """A chapter with its subsections."""
    chapter_no: int
    chapter_title: str
    chunks: list[Chunk]


def group_chunks_by_chapter(
    chunks: list[Chunk],
    min_chapter_score: float = 5.0,
) -> list[ChapterGroup]:
    """Group chunks into chapters. H2 with score >= min_chapter_score starts a new chapter.
    
    H3 chunks and H2 chunks below threshold get grouped under the nearest preceding chapter.
    """
    groups: list[ChapterGroup] = []
    current: ChapterGroup | None = None
    
    for chunk in chunks:
        if chunk.level == 2 and chunk.score >= min_chapter_score:
            # New chapter
            current = ChapterGroup(
                chapter_no=chunk.chapter_no,
                chapter_title=chunk.heading,
                chunks=[chunk],
            )
            groups.append(current)
        elif current is not None:
            # Subsection or low-score heading — attach to current chapter
            current.chunks.append(chunk)
    
    return groups
```

```python
"""Document profile detection for PDF processing."""

from pathlib import Path
import fitz
import re


def detect_profile(path: Path) -> dict:
    """Analyze PDF and return a document profile dict.
    
    Returns:
        type: born-digital | ocr-needed | diagram-heavy | toc-heavy | cjk-broken
        body_size: estimated body font size
        encoding_ok: True if text extraction produces readable text
        page_count: number of pages
        has_index: True if document has Index section
    """
    doc = fitz.open(str(path))
    page_count = len(doc)
    
    # Body size
    from collections import Counter
    sizes = []
    for page in doc:
        blocks = page.get_text("dict")["blocks"]
        for block in blocks:
            if block["type"] == 0:
                for line in block["lines"]:
                    for span in line["spans"]:
                        sizes.append(span["size"])
        if len(sizes) > 200:
            break
    
    body_size = 12.0
    if sizes:
        sizes.sort()
        body_size = sizes[len(sizes) // 2]
    
    # Check for encoding issues: sample some text from first 5 pages
    garbled_chars = 0
    total_chars = 0
    for i in range(min(5, len(doc))):
        text = doc[i].get_text()
        for ch in text:
            if ord(ch) > 127 and not (0x4E00 <= ord(ch) <= 0x9FFF):  # not CJK
                garbled_chars += 1
            total_chars += 1
    
    encoding_ok = total_chars == 0 or (garbled_chars / max(total_chars, 1)) < 0.3
    
    # Check for many drawings (diagram-heavy)
    drawing_count = 0
    for i in range(min(10, len(doc))):
        drawings = doc[i].cluster_drawings() if hasattr(doc[i], "cluster_drawings") else []
        drawing_count += len(drawings)
    
    # Check for Index
    has_index = False
    for i in range(max(0, len(doc) - 20), len(doc)):
        text = doc[i].get_text("text")
        if re.search(r"^Index\s*$", text, re.MULTILINE):
            has_index = True
            break
    
    doc.close()
    
    # Determine type
    if not encoding_ok:
        doc_type = "cjk-broken"
    elif drawing_count > 50:
        doc_type = "diagram-heavy"
    elif page_count > 500:
        doc_type = "born-digital"
    else:
        doc_type = "born-digital"
    
    return {
        "type": doc_type,
        "body_size": body_size,
        "encoding_ok": encoding_ok,
        "page_count": page_count,
        "has_index": has_index,
    }
```

```python
def convert(
    input: Path = typer.Argument(..., help="Input PDF path"),
    output: Path = typer.Option("./output", "-o", "--output", help="Output directory"),
    domain: list[str] = typer.Option([], "--domain", help="Domain tags"),
    topic: list[str] = typer.Option([], "--topic", help="Topic tags"),
    edition: str = typer.Option("", "--edition", help="Edition string"),
    dry_run: bool = typer.Option(False, "--dry-run", help="Show structure only"),
):
    """Convert PDF to structured markdown chunks."""
    input_path = Path(input)
    output_path = Path(output)
    
    # Metadata
    meta = get_metadata(input_path)
    meta["domains"] = domain
    meta["topics"] = topic
    meta["edition"] = edition or None
    book_slug = meta.get("id") or slugify(input_path.stem)
    
    # Document profile
    profile = detect_profile(input_path)
    if not profile["encoding_ok"]:
        typer.echo(f"Warning: Encoding issues detected — consider OCR pipeline", err=True)
    
    # Extract
    chunks = extract_pdf(input_path)
    
    # Filter
    chunks = filter_toc_chunks(chunks)
    chunks = [c for c in chunks if not (c.near_drawing and c.score < 3)]
    
    # Score each chunk
    body_size = profile["body_size"]
    for chunk in chunks:
        chunk.score = score_heading(chunk, body_size)
    
    if dry_run:
        _show_dry_run(chunks, profile)
        return
    
    # Group and write
    groups = group_chunks_by_chapter(chunks)
    write_chapter_groups(output_path, groups, meta, book_slug)
    typer.echo(f"Written {sum(len(g.chunks) for g in groups)} chunks to {output_path / book_slug}")
```
